A total of 28,643 Ensembl IDs were mapped to pathways from the TFT gene sets.

formatP <- function(p, m = 0.0001){
out <- rep("", length(p))
out[p < m] <- sprintf("%.2e", p[p<m])
out[p >= m] <- sprintf("%.4f", p[p>=m])
out <- as.numeric(out)
out
}
